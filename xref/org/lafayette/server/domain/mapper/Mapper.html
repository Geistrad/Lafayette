15  import org.lafayette.server.domain.DomainObject;

17  /**
18   * Interface for mapper which maps {@link DomainObject domain objects} to the database.
19   *
20   * @param <T> concrete type of domain object
21   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
22   */
23  public interface Mapper<T extends DomainObject> {
24  
25      /**
26       * Insert domain object into database.
27       *
28       * @param subject domain object to insert
29       * @return primary key id of inserted record set
30       */
31      int insert(T subject);
32      /**
33       * updates domain object from database.
34       *
35       * @param subject domain object to update
36       */
37      void update(final T subject);
38      /**
39       * Deletes domain object from database.
40       *
41       * @param subject domain object to delete
42       */
43      void delete(final T subject);
44  
45  }
