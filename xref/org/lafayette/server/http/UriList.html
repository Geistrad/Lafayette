12  package org.lafayette.server.http;
13  
14  import java.net.URI;
15  import java.util.TreeSet;
16  import org.apache.commons.lang3.Validate;
17  
18  /**
19   * Represents a set of URIs used for responses of {@link MediaType#TEXT_URI_LIST type URI-list}.
20   *
21   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
22   */
23  public class UriList extends TreeSet<URI> implements Typeable {
24  
25      /**
26       * Optional comment.
27       */
28      private String comment = "";
29  
30      /**
31       * Set optional comment.
32       *
33       * Will be placed before the URI list preceeded with "# ".
34       *
35       * @param comment must not be {@code null}.
36       */
37      public void setComment(final String comment) {
38          Validate.notNull(comment, "Comment must not be null!");
39          this.comment = comment;
40      }
41  
42      @Override
43      public String getMediaType() {
44          return MediaType.TEXT_URI_LIST;
45      }
46  
47      @Override
48      public String toString() {
49          final StringBuilder buffer = new StringBuilder();
50  
51          if (!comment.isEmpty()) {
52              buffer.append("# ").append(comment).append(Constants.NL);
53          }
54  
55          for (final URI uri : this) {
56              buffer.append(uri).append(Constants.NL);
57          }
58  
59          return buffer.toString();
60      }
61  
62  }
