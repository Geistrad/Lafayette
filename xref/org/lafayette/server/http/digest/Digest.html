12  package org.lafayette.server.http.digest;
13  
14  import org.apache.commons.codec.digest.DigestUtils;
15  
16  /**
17   * Helper to generate HTTP digest authentication responses.
18   *
19   * Format is:
20   * <pre>
21   * $RESPONSE = md5(
22   *                  md5( username + ":" + realm + ":" + password ) +    // User data.
23   *                  ":" + nonce + ":" +
24   *                  md5( httpMethod + ":" + requestedUri )              // Request data.
25   *             )
26   * </pre>
27   *
28   * Server response header:
29   * <pre>
30   * HTTP/1.1 401 Unauthorized
31   * WWW-Authenticate: Digest real="Private Area", nonce="IrTfjizEdXmIdlwHwkDJx0"
32   * ...
33   * </pre>
34   *
35   * Client Request:
36   * <pre>
37   * GET / HTTP/1.1
38   * Authorization: Digest username="Foo",
39   *                       realm="Private Area",
40   *                       nonce="IrTfjizEdXmIdlwHwkDJx0",
41   *                       uri="/",
42   *                       response="$RESPONSE"
43   * ...
44   * </pre>
45   *
46   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
47   */
48  public final class Digest {
49  
50      private Digest() {
51          super();
52      }
53  
54      public static String digestUserData(final String username, final String password, final String realm) {
55          return DigestUtils.md5Hex(String.format("%s:%s:%s", username, realm, password));
56      }
57  
58      public static String digestRequestData(final String httpMethod, final String requestedUri) {
59          return DigestUtils.md5Hex(String.format("%s:%s", httpMethod, requestedUri));
60      }
61  
62      public static String digest(final String userData, final String nonce, final String requetsData) {
63          return DigestUtils.md5Hex(String.format("%s:%s:%s", userData, nonce, requetsData));
64      }
65  
66      public static String digest(final ResponseParameters params) {
67          final String userData = digestUserData(params.getUsername(), params.getPassword(), params.getRealm());
68          final String requestData = digestRequestData(params.getHttpMethod(), params.getRequestedUri());
69          return digest(userData, params.getNonce(), requestData);
70      }
71  
72  }
