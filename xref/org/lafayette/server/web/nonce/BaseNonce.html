137     protected String digest(final String plaintext) {
138         switch (digestAlgorithm) {
139             case MD5:
140                 return DigestUtils.md5Hex(plaintext);
141             case SHA1:
142                 return DigestUtils.sha1Hex(plaintext);
143             default:
144                 throw new IllegalArgumentException(String.format("Unsupported algorithm '%s'!", digestAlgorithm));
145         }
146     }

148     @Override

152         nonceString.append(getSalt()).append(Long.toHexString(nonceVal));
153         return digest(nonceString.toString());
