30  @Title("Sign up")

32  
33      /* User interface components are stored in session. */
34      private final Button createAccountButton = new Button("Create Account");
35  
36      private final TextField loginNameField = new TextField("Login name");
37      private final TextField emailAddressField = new TextField("Email address");
38      private final TextField passwordField = new TextField("Password");
39  
40      /**
41       * After UI class is created, init() is executed.
42       *
43       * You should build and wire up your user interface here.
44       * @param request
45       */
46      @Override
47      protected void init(final VaadinRequest request) {
48          initLayout();
49          initCreateAccountButton();
50      }
51  
52      /**
53       * Initializes layout and components.
54       */
55      private void initLayout() {
56          final VerticalLayout mainLayout = new VerticalLayout();
57          mainLayout.setSizeFull();
58          setContent(mainLayout);
59          final FormLayout formLayout = new FormLayout();
60          formLayout.setSizeUndefined();
61  
62          formLayout.addComponent(loginNameField);
63          loginNameField.setRequired(true);
64          loginNameField.setRequiredError("The Field may not be empty.");
65          formLayout.addComponent(emailAddressField);
66          emailAddressField.setRequired(true);
67          emailAddressField.setRequiredError("The Field may not be empty.");
68          formLayout.addComponent(passwordField);
69          passwordField.setRequired(true);
70          passwordField.setRequiredError("The Field may not be empty.");
71          formLayout.addComponent(createAccountButton);
72  
73          mainLayout.addComponent(formLayout);
74          mainLayout.setComponentAlignment(formLayout, Alignment.TOP_CENTER);
75      }
76  
77      private void initCreateAccountButton() {
78          createAccountButton.addClickListener(new ClickListener() {
79              public void buttonClick(final ClickEvent event) {
80              }
81          });
82      }
