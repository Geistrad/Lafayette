39  @Title("Sign up")
40  public class RegisrationUI extends UI {
41      private static final String ERR_MSG_PASSWORD_MUST_NOT_BE_EMPTY = "Password must not be empty!";
42      private static final String ERR_MSG_EMAIL_ADDRESS_MUST_NOT_BE_EMPTY = "Email address must not be empty!";
43      private static final String ERR_MSG_LOGIN_NAME_MUST_NOT_BE_EMPTY = "Login name must not be empty!";
44      private static final String ERR_MSG_PASSWORD_MUST_NOT_BE_SAME_AS_LOGIN_NAME = "Password must not be same as login name!";
45      private static final String ERR_MSG_PASSWORD_MUST_NOT_BE_SAME_AS_EMAIL_ADDRES = "Password must not be same as email address!";
46      private static final String ERR_MSG_PASSWORD_MUST_HAVE_AT_LEAST_8_CHARATCERS = "Password must have at least 8 charatcers!";

48      private final Button createAccountButton = new Button("Create Account");
49      private final TextField loginNameField = new TextField("Login name");
50      private final TextField emailAddressField = new TextField("Email address");
51      private final TextField passwordField = new TextField("Password");

94          formLayout.addComponent(emailAddressField);
95

106     private void initCreateAccountButton() {
107         createAccountButton.addClickListener(new ClickListener() {
108             @Override
109             public void buttonClick(final ClickEvent event) {
110                 boolean isInputFaulty = false;
111 
112                 final String loginName = loginNameField.getValue();
113                 final String password = passwordField.getValue();
114                 final String emailAddress = emailAddressField.getValue();
115 
116 
117                 try {
118                     validatePassword(password, loginName, emailAddress);
119                 } catch (IllegalArgumentException ex) {
120                     passwordField.setComponentError(new UserError(ex.getMessage()));
121                     isInputFaulty = true;
122                 }
123 
124                 try {
125                     validateEmailAddress(emailAddress);
126                 } catch (IllegalArgumentException ex) {
127                     emailAddressField.setComponentError(new UserError(ex.getMessage()));
128                     isInputFaulty = true;
129                 }
130 
131                 if (isInputFaulty) {
132                     return;
133                 }
134 
135                 try {
136                     registration.registerNewUser(
137                             loginName,
138                             password,
139                             registry().getInitParameters().getRealm(),
140                             emailAddress);
141                 } catch (ServiceExcpetion ex) {
142                     Notification.show("Error!", ex.getMessage(), Notification.Type.ERROR_MESSAGE);
143                 }
144             }
145         });
146     }
147 
148     static void validatePassword(final String password, final String loginName, final String emailAdress) {
149         if (password.equalsIgnoreCase(loginName)) {
150             throw new IllegalArgumentException(ERR_MSG_PASSWORD_MUST_NOT_BE_SAME_AS_LOGIN_NAME);
151         }
152 
153         if (password.equalsIgnoreCase(emailAdress)) {
154             throw new IllegalArgumentException(ERR_MSG_PASSWORD_MUST_NOT_BE_SAME_AS_EMAIL_ADDRES);
155         }
156 
157         if (password.length() < 8) {
158             throw new IllegalArgumentException(ERR_MSG_PASSWORD_MUST_HAVE_AT_LEAST_8_CHARATCERS);
159         }
160     }
161 
162     static void validateEmailAddress(final String emailaddress) {
163         if (!EmailValidator.getInstance().isValid(emailaddress)) {
164             throw new IllegalArgumentException("Email address is not valid!");
165         }
166     }
