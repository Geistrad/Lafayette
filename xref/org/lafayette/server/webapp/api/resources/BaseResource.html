39  /**
40   * Common functionality for all resource classes.
41   *
42   * TODO Write tests for this class.
43   *
44   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
45   */
46  abstract class BaseResource implements AbstractResourceModelListener {
47  
48      /**
49       * Logger facility.
50       */
51      protected final Logger log = Log.getLogger(this);
52      /**
53       * List of URIs available from a resource's index.
54       */
55      private final UriList indexUriList;
56      /**
57       * Used to format message pack format.
58       */
59      private final MessagePack msgpack = new MessagePack();
60      /**
61       * Servlet context injected by Jersey.
62       */
63      @Context private ServletContext servlet;
64      /**
65       * Security context injected by Jersey.
66       */
67      @Context private HttpHeaders headers;
68      /**
69       * URI info injected by Jersey.
70       */
71      @Context private UriInfo uriInfo;
72      /**
73       * Security context injected by Jersey.
74       */
75      @Context private SecurityContext security;
76  
77      /**
78       * Dedicated constructor.
79       */
80      public BaseResource() {
81          super();
82          indexUriList = new UriList();
83          indexUriList.setComment("# Available URIs:");
84      }
85  
86      @Override
87      //CHECKSTYLE:OFF
88      public void onLoaded(final AbstractResourceModelContext modelContext) {
89      //CHECKSTYLE:ON
90          init();
91      }
92  
93      /**
94       * Initialize resource.
95       *
96       * - call {@link #addUrisToIndexList(org.lafayette.server.http.UriList)}
97       */
98      private void init() {
99          try {
100             addUrisToIndexList(indexUriList);
101         } catch (URISyntaxException ex) {
102             log.fatal("Error adding index uris: %s", ex);
103         }
104     }
105 
106     /**
107      * Add all URIs to all sub resources to the given list.
108      *
109      * @param list list to add URIs
110      * @throws URISyntaxException if URI has bad syntax
111      */
112     protected abstract void addUrisToIndexList(UriList list) throws URISyntaxException;
113 
114     /**
115      * Get object registry for shared objects.
116      *
117      * @return never {@code null}
118      */
119     protected Registry registry() {
120         return (Registry) servlet.getAttribute(ServerContextListener.REGISRTY);
121     }
122 
123     /**
124      * Get the servlet context.
125      *
126      * @return never {@code null}
127      */
128     protected ServletContext servlet() {
129         return servlet;
130     }
131 
132     /**
133      * Get the requests HTTP headers.
134      *
135      * @return never {@code null}
136      */
137     protected HttpHeaders headers() {
138         return headers;
139     }
140 
141     /**
142      * Get the requests URI info.
143      *
144      * @return never {@code null}
145      */
146     protected UriInfo uriInfo() {
147         return uriInfo;
148     }
149 
150     /**
151      * Get the requests security context.
152      *
153      * @return never {@code null}
154      */
155     protected SecurityContext security() {
156         return security;
157     }
158 
159     protected Services services() {
160         return registry().getServices();
161     }
162 
163     /**
164      * Create an 500 error response.
165      *
166      * @param message error message
167      * @return never {@code null}
168      */
169     protected Response createServerErrorResponse(final String message) {
170         return Response.serverError()
171                 .entity(message)
172                 .build();
173     }
174 
175     /**
176      * Create an 400 error response.
177      *
178      * @param message error message
179      * @return never {@code null}
180      */
181     protected Response createClientErrorResponse(final String message) {
182         return Response.status(Response.Status.BAD_REQUEST)
183                 .entity(message)
184                 .build();
185     }
186 
187     /**
188      * Throws a error exception with status code 500.
189      *
190      * @param message error message
191      */
192     protected void raiseServerError(final String message) {
193         throw new WebApplicationException(createServerErrorResponse(message));
194     }
195 
196     /**
197      * Throws a error exception with status code 400.
198      *
199      * @param message error message
200      */
201     protected void raiseClientError(final String message) {
202         throw new WebApplicationException(createClientErrorResponse(message));
203     }
204 
205     /**
206      * Throw a 404 error.
207      *
208      * @param resource resource name
209      * @param id id of resource
210      * @throws NotFoundException always
211      */
212     protected void raiseIdNotFoundError(final String resource, final String id) throws NotFoundException {
213         final String message = String.format("Can't find '%s' with id '%s'!", resource, id);
214         throw new NotFoundException(message);
215     }
216 
217     /**
218      * Throws a web application error for a missing request property.
219      *
220      * @param name of missing property
221      * @throws WebApplicationException always
222      */
223     protected void raiseMissingPropertyError(final String name) throws WebApplicationException {
224         final String message = String.format("Property '%s' missing!", name);
225         throw new WebApplicationException(createServerErrorResponse(message));
226     }
227 
228     /**
229      * Returns message pack marshalled domain object.
230      *
231      * @param object to marshal, must not be {@code null}
232      * @return message pack formated bytes
233      */
234     protected byte[] formatMessagePack(final DomainObject object) {
235         Validate.notNull(object);
236         try {
237             return msgpack.write(object);
238         } catch (IOException ex) {
239             log.fatal("Can not marshall object to message pack: %s!", object);
240             return new byte[] {};
241         }
242     }
243 
244     /**
245      * Returns the resource index as plain text.
246      *
247      * @return never {@code null}
248      */
249     @GET
250     @Produces(MediaType.TEXT_URI_LIST)
251     public String indexAsUriList() {
252         log.info("Respond with URI list for %s", servlet.getRealPath(""));
253         return indexUriList.toString();
254     }
