24  /**
25   * Filters the request/response for authentication.
26   *
27   * Filter is only applied to resource methods with annotation {@link Authentication}.
28   *
29   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
30   */
31  @Provider
32  public class AuthenticationResourceFilterFactory implements ResourceFilterFactory {
33  
34      @Context
35      private ServletContext servlet;
36      @Context
37      private HttpHeaders headers;
38  
39  //    @Inject
40  //    public AuthenticationResourceFilterFactory(final SecuirityContextFilter security) {
41  //        super();
42  //        this.security = security;
43  //    }
44  
45      @Override
46      public List<ResourceFilter> create(final AbstractMethod am) {
47          if (am.isAnnotationPresent(Authentication.class)) {
48              return Collections.<ResourceFilter>singletonList(new SecurityContextFilterDigest(servlet, headers));
49          }
50  
51          return Collections.emptyList();
52      }
53  
54  }
