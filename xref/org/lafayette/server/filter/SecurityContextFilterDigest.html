39  /**
40   * Filters resource request/response for digest authentication.
41   *
42   * If a user is authorized then a {@link User} representing the user is injected as {@link Principal} to the
43   * {@link SecurityContext}.
44   *
45   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
46   */
47  @Provider
48  class SecurityContextFilterDigest implements SecuirityContextFilter {
49  
50      /**
51       * Logger facility.
52       */
53      private final Logger log = Log.getLogger(this);
54      /**
55       * The servlet context to get registry from.
56       */
57      private final ServletContext servlet;
58      /**
59       * The request's HTTP headers.
60       */
61      private final HttpHeaders headers;
62  
63      /**
64       * Dedicated constructor.
65       *
66       * @param servlet must not be {2code null}
67       * @param headers must not be {2code null}
68       */
69      public SecurityContextFilterDigest(final ServletContext servlet, final HttpHeaders headers) {
70          super();
71          Validate.notNull(servlet, "Servlet context must not be null!");
72          this.servlet = servlet;
73          Validate.notNull(headers, "HTTP headers must not be null!");
74          this.headers = headers;
75      }
76  
77      /**
78       * Get global registry from servlet context.
79       *
80       * @return always same instance
81       */
82      Registry registry() {
83          return (Registry) servlet.getAttribute(ServerContextListener.REGISRTY);
84      }
85  
86      @Override
87      public ContainerRequestFilter getRequestFilter() {
88          return this;
89      }
90  
91      @Override
92      public ContainerResponseFilter getResponseFilter() {
93          return this;
94      }
95  
96      @Override
97      public ContainerRequest filter(final ContainerRequest request) {
98          log.debug("Start authentication");
99          final String header = getAuthorizationHeader();
100 
101         if (header.isEmpty()) {
102             log.debug("Request doesn't have authentication header! Send authentication response.");
103             final ResponseParameters params = new ResponseParameters();
104             params.setRealm(registry().getServerConfig().getSecurotyRealm());
105             params.setNonce(registry().getNongeGenerator().getNext());
106             throw new UnauthorizedException(params);
107         } else {
108             log.debug("Verify authentication.");
109             final RequestParameters params = AuthorizationHeaderParser.parseDigestHeaderValue(header);
110 
111             if (verifyAuthentiaction(params)) {
112                 createAndSetPrincipal(request, params);
113                 return request;
114             } else {
115                 log.debug("Not authorized! Send forbidden response.");
116                 throw new ForbiddenException();
117             }
118         }
119     }
120 
121     @Override
122     public ContainerResponse filter(final ContainerRequest request, final ContainerResponse response) {
123         return response;
124     }
125 
126     /**
127      * Extracts the authorization header from the requests headers.
128      *
129      * @return may return empty string, but never {@code null}
130      */
131     private String getAuthorizationHeader() {
132         final List<String> authHeader = headers.getRequestHeader(HttpHeaders.AUTHORIZATION);
133 
134         if (authHeader == null || authHeader.isEmpty()) {
135             log.debug("No authorization header sent by client.");
136             return "";
137         }
138 
139         if (authHeader.size() > 1) {
140             log.warn("More than one authorization header sent by client! Using first one and ignore others.");
141         }
142 
143         return authHeader.get(0);
144     }
145 
146     private boolean verifyAuthentiaction(final RequestParameters params) {
147         final User principal = findPrincipal(params);
148 
149         if (null == principal) {
150             return false;
151         }
152 
153         // FIXME Verify nonce
154         final String expectedResponse = Digest.digest(principal.getHashedUserData(),
155                 params.getNonce(),
156                 Digest.digestRequestData(params.getHttpMethod(), params.getRequestedUri()));
157         return expectedResponse.equalsIgnoreCase(params.getResponse());
158     }
159 
160     private void createAndSetPrincipal(final ContainerRequest request, final RequestParameters params) {
161         final User principal = findPrincipal(params);
162         final Collection<Role> roles = registry().getMappers()
163                 .createRoleMapper()
164                 .findByUserId(principal.getId());
165         principal.addAllRoles(roles);
166         request.setSecurityContext(new SecurityContextImpl(principal, request.isSecure()));
167     }
168 
169     private User findPrincipal(final RequestParameters params) {
170         return registry().getMappers()
171                 .createUserMapper()
172                 .findByLoginName(params.getUsername());
173     }
