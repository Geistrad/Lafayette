12  package org.lafayette.server;
13  
14  /**
15   * Defines the stage in which the server runs.
16   *
17   * Supported stages at the moment are:
18   * <ul>
19   * <li>{@link Stages#DEVELOPMENT development stage}
20   * <li>{@link Stages#PRODUCTION production stage}
21   * </ul>
22   *
23   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
24   */
25  public class Stage {
26  
27      /**
28       * Default stage used as fallback is {@link Stages#PRODUCTION}.
29       */
30      private static final Stages DEFUALT_STAGE = Stages.PRODUCTION;
31      /**
32       * Current stage.
33       */
34      private final Stages stage;
35  
36      /**
37       * Convenience constructor which initializes the current stage with {@link #DEFUALT_STAGE}.
38       */
39      public Stage() {
40          this(DEFUALT_STAGE);
41      }
42  
43      /**
44       * Convenience constructor which maps given string to a {@link Stages stage}
45       * via {@link Stages#map(java.lang.String)}.
46       *
47       * @param stageFromEnviroment mapped environment name
48       */
49      public Stage(final String stageFromEnviroment) {
50          this(Stages.map(stageFromEnviroment));
51      }
52  
53      /**
54       * Dedicated constructor.
55       *
56       * @param s current stage
57       */
58      public Stage(final Stages s) {
59          super();
60          stage = s;
61      }
62  
63      /**
64       * Get the current stage.
65       *
66       * @return never {@code null}
67       */
68      public Stages getStage() {
69          return stage;
70      }
71  
72      @Override
73      public String toString() {
74          return stage.toString();
75      }
76  
77      /**
78       * Supported stages.
79       */
80      public static enum Stages {
81  
82          /**
83           * Development stage.
84           */
85          DEVELOPMENT,
86          /**
87           * Production stage.
88           */
89          PRODUCTION;
90  
91          @Override
92          public String toString() {
93              return name();
94          }
95  
96          /**
97           * Try to find an according {@link Stages stage} for a given name.
98           *
99           * Supported names are:
100          * <ul>
101          * <li>development
102          * <li>production
103          * </ul>
104          *
105          * @param name case does not matter
106          * @return never returns {@code null}, if no proper stage was found by the name {@link Stage#DEFUALT_STAGE}
107          *         will be returned
108          */
109         static Stages map(final String name) {
110             if ("development".equalsIgnoreCase(name)) {
111                 return DEVELOPMENT;
112             } else if ("production".equalsIgnoreCase(name)) {
113                 return PRODUCTION;
114             }
115 
116             return DEFUALT_STAGE;
117         }
118     }
119 }
