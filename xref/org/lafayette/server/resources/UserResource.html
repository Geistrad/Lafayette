35   * Serves the user resource.

37   * @author Sven Strittmatter <weltraumschaf@googlemail.com>

39  @Path("/user")
40  public class UserResource extends BaseResource {

42      private static final int PAD_ID = 5;
43      private static final int PAD_LOGIN_NAME = 16;
44      private static final int PAD_HASHED_USER_DATA = 34;
45      private static final int LIMIT = 25;
46  
47      @Override
48      protected void addUrisToIndexList(final UriList indexUriList) throws URISyntaxException {
49          final UriBuilder ub = uriInfo().getAbsolutePathBuilder();
50          ub.path("/{id}");
51          indexUriList.add(ub.build());
52      }

54      @GET
55      @Produces(MediaType.TEXT_PLAIN)
56      public String usersAsText() {
57          final StringBuilder buffer = new StringBuilder();
58          final UserFinder finder = finders().forUsers();
59          buffer.append(StringUtils.rightPad("id", PAD_ID))
60                  .append(StringUtils.rightPad("loginName", PAD_LOGIN_NAME))
61                  .append(StringUtils.rightPad("hashedUserData", PAD_HASHED_USER_DATA))
62                  .append("salt")
63                  .append(Constants.NL);
64  
65          for (final User user : finder.findAll(LIMIT, 0)) {
66              buffer.append(StringUtils.rightPad(String.valueOf(user.getId()), PAD_ID))
67                      .append(StringUtils.rightPad(user.getLoginName(), PAD_LOGIN_NAME))
68                      .append(StringUtils.rightPad(user.getHashedUserData(), PAD_HASHED_USER_DATA))
69                      .append(Constants.NL);
70          }
71  
72          return buffer.toString();
73      }

75      @GET
76      @Path("/{id}")
77      @Produces(MediaType.TEXT_PLAIN)
78      public String userAsPlainText(@PathParam("id") String id) {
79          final User user = findUserById(id);
80  
81          if (null == user) {
82              raiseIdNotFoundError("user", id);
83          }
84  
85          return user.toString();
86      }

101     @GET
102     @Path("/{id}")
103     @Produces(MediaType.APPLICATION_X_MSGPACK)
104     public Response userAsMessagePack(@PathParam("id") String id) throws IOException {
105         final User user = findUserById(id);
106 
107         if (null == user) {
108             raiseIdNotFoundError("user", id);
109         }
110 
111         return Response.ok(formatMessagePack(user), MediaType.APPLICATION_X_MSGPACK).build();
112     }

114     @PUT
115     @Consumes(MediaType.APPLICATION_JSON)
116     public Response createUser(JSONObject jsonEntity) throws JSONException, Exception {
117         if (!jsonEntity.has("loginName")) {
118             raiseMissingPropertyError("loginName");
119         }
120 
121         final String loginName = jsonEntity.getString("loginName");
122         final User newUser = new User(loginName, ""); // FIXME Set hashed user data (password).
123         registry().getMappers().createUserMapper().insert(newUser);
124         final URI uri = uriInfo().getAbsolutePathBuilder()
125                 .path(String.valueOf(newUser.getId()))
126                 .build();
127 
128         return Response.created(uri)
129                 .entity(newUser)
130                 .build();
131     }

133     /**
134      * Find a user by it's id.
135      *
136      * @param id user's id
137      * @return may be {@code null}, if user was not found
138      * throws NumberFormatException if passed in user id is not parseable to integer
139      */
140     private User findUserById(String id) {
141         final UserFinder finder = finders().forUsers();
142         return finder.find(Integer.parseInt(id));
143     }
