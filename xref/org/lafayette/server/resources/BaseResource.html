37  /**
38   * Common functionality for all resource classes.
39   *
40   * TODO Write tests for this class.
41   *
42   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
43   */
44  abstract class BaseResource {
45  
46      /**
47       * Logger facility.
48       */
49      protected final Logger log = Log.getLogger(this);
50      /**
51       * List of URIs available from a resource's index.
52       */
53      private final UriList indexUriList;
54      /**
55       * Used to format message pack format.
56       */
57      private final MessagePack msgpack = new MessagePack();
58      /**
59       * Servlet context injected by Jersey.
60       */
61      @Context private ServletContext servlet;
62      /**
63       * Security context injected by Jersey.
64       */
65      @Context private HttpHeaders headers;
66      /**
67       * URI info injected by Jersey.
68       */
69      @Context private UriInfo uriInfo;
70      /**
71       * Security context injected by Jersey.
72       */
73      @Context private SecurityContext security;
74      /**
75       * Finder factory.
76       *
77       * Lazy computed.
78       */
79      private Finders finders;
80  
81      /**
82       * Dedicated constructor.
83       */
84      public BaseResource() {
85          super();
86          indexUriList = new UriList();
87          indexUriList.setComment("# Available URIs:");
88          init();
89      }
90  
91      /**
92       * Initialize resource.
93       *
94       * - call {@link #addUrisToIndexList(org.lafayette.server.http.UriList)}
95       */
96      private void init() {
97          try {
98              addUrisToIndexList(indexUriList);
99          } catch (URISyntaxException ex) {
100             log.fatal("Error adding index uris: %s", ex);
101         }
102     }
103 
104     /**
105      * Add all URIs to all sub resources to the given list.
106      *
107      * @param list list to add URIs
108      * @throws URISyntaxException if URI has bad syntax
109      */
110     protected abstract void addUrisToIndexList(UriList list) throws URISyntaxException;
111 
112     /**
113      * Get object registry for shared objects.
114      *
115      * @return never {@code null}
116      */
117     protected Registry registry() {
118         return (Registry) servlet.getAttribute(ServerContextListener.REGISRTY);
119     }
120 
121     /**
122      * Get the servlet context.
123      *
124      * @return never {@code null}
125      */
126     protected ServletContext servlet() {
127         return servlet;
128     }
129 
130     /**
131      * Get the requests HTTP headers.
132      *
133      * @return never {@code null}
134      */
135     protected HttpHeaders headers() {
136         return headers;
137     }
138 
139     /**
140      * Get the requests URI info.
141      *
142      * @return never {@code null}
143      */
144     protected UriInfo uriInfo() {
145         return uriInfo;
146     }
147 
148     /**
149      * Get the requests security context.
150      *
151      * @return never {@code null}
152      */
153     protected SecurityContext security() {
154         return security;
155     }
156 
157     /**
158      * Get domain object finders.
159      *
160      * @return never {@code null} and same instance
161      */
162     protected synchronized Finders finders() {
163         if (null == finders) {
164             finders = new Finders(registry().getMappers());
165         }
166 
167         return finders;
168     }
169 
170     /**
171      * Create an 500 error response.
172      *
173      * @param message error message
174      * @return never {@code null}
175      */
176     protected Response createServerErrorResponse(final String message) {
177         return Response.serverError()
178                 .entity(message)
179                 .build();
180     }
181 
182     /**
183      * Create an 400 error response.
184      *
185      * @param message error message
186      * @return never {@code null}
187      */
188     protected Response createClientErrorResponse(final String message) {
189         return Response.status(Response.Status.BAD_REQUEST)
190                 .entity(message)
191                 .build();
192     }
193 
194     /**
195      * Throws a error exception with status code 500.
196      *
197      * @param message error message
198      */
199     protected void raiseServerError(final String message) {
200         throw new WebApplicationException(createServerErrorResponse(message));
201     }
202 
203     /**
204      * Throws a error exception with status code 400.
205      *
206      * @param message error message
207      */
208     protected void raiseClientError(final String message) {
209         throw new WebApplicationException(createClientErrorResponse(message));
210     }
211 
212     /**
213      * Throw a 404 error.
214      *
215      * @param resource resource name
216      * @param id id of resource
217      * @throws NotFoundException always
218      */
219     protected void raiseIdNotFoundError(final String resource, final String id) throws NotFoundException {
220         final String message = String.format("Can't find '%s' with id '%s'!", resource, id);
221         throw new NotFoundException(message);
222     }
223 
224     /**
225      * Throws a web application error for a missing request property.
226      *
227      * @param name of missing property
228      * @throws WebApplicationException always
229      */
230     protected void raiseMissingPropertyError(final String name) throws WebApplicationException {
231         final String message = String.format("Property '%s' missing!", name);
232         throw new WebApplicationException(createServerErrorResponse(message));
233     }
234 
235     /**
236      * Returns message pack marshalled domain object.
237      *
238      * @param object to marshal, must not be {@code null}
239      * @return message pack formated bytes
240      */
241     protected byte[] formatMessagePack(final DomainObject object) {
242         Validate.notNull(object);
243         try {
244             return msgpack.write(object);
245         } catch (IOException ex) {
246             log.fatal("Can not marshall object to message pack: %s!", object);
247             return new byte[] {};
248         }
249     }
250 
251     /**
252      * Returns the resource index as plain text.
253      *
254      * @return never {@code null}
255      */
256     @GET
257     @Produces(MediaType.TEXT_URI_LIST)
258     public String indexAsUriList() {
259         log.info("Respond with URI list for %s", servlet.getRealPath(""));
260         return indexUriList.toString();
261     }
